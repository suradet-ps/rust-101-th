# Rust-101, ส่วนที่ 08: ชนิดข้อมูลสมทบ (Associated Types), โมดูล

```rust
use std::{cmp, ops};
use part05::BigInt;
```

เป็นเป้าหมายถัดไปของเรา เราจะมาอิมพลีเมนท์การบวกให้กับ `BigInt` ของเรา ประเด็นหลักตรงนี้คือการจัดการกับผลลัพธ์ล้นหลัก (overflow) ก่อนอื่น เราจะต้องตรวจจับเมื่อ overflow เกิดขึ้น สิ่งนี้ถูกเก็บในสิ่งที่เรียกว่าตัวทด (carry bit) และเราต้องนำข้อมูลนี้ไปรวมกับหลักเลขคู่ถัดไปที่เราบวก ปฐมบทหลักของการบวกจึงคือการบวกเลขโดดสองตัวและตัวทดหนึ่งตัว และคืนค่าเลขโดดผลรวมและตัวทดถัดไป

ดังนั้น เรามาเขียนฟังก์ชันเพื่อ "บวกแบบมีตัวทด" กัน และกำหนดชนิดข้อมูลที่เหมาะสมให้มัน สังเกตการรองรับคู่ลำดับ (pairs) ที่มีมาให้ใน Rust

```rust
fn overflowing_add(a: u64, b: u64, carry: bool) -> (u64, bool) {
    // จุดยืนของ Rust ต่อ integer overflows อาจจะน่าประหลาดใจสักหน่อย: โดยทั่วไป เมื่อเราเขียน a + b
    // การเกิด overflow ถือว่าเป็นข้อผิดพลาด ถ้าคุณคอมไพล์โปรแกรมของคุณในโหมดดีบัก Rust จะ
    // ตรวจสอบข้อผิดพลาดนั้นจริงๆ และแพนิคโปรแกรมในกรณีที่มี overflows ด้วยเหตุผลด้านประสิทธิภาพ
    // การตรวจสอบดังกล่าวจึงไม่ถูกใส่ไว้ใน build สำหรับ release สาเหตุของเรื่องนี้คือ
    // ช่องโหว่ความปลอดภัยร้ายแรงจำนวนมากมีสาเหตุมาจาก integer overflows ดังนั้นการแค่
    // ทึกทักเอา "เป็นค่าตั้งต้น" ว่าพวกมันเป็นความตั้งใจ จึงเป็นเรื่องอันตราย
    //
    // ถ้าคุณต้องการให้ overflow เกิดขึ้นอย่างชัดแจ้ง คุณสามารถเรียกฟังก์ชัน wrapping_add
    // (ดูเอกสารประกอบ มีฟังก์ชันคล้ายกันสำหรับปฏิบัติการทางคณิตศาสตร์อื่นๆ)
    // ยังมีฟังก์ชันที่คล้ายกันอย่าง checked_add ฯลฯ เพื่อบังคับใช้การตรวจสอบ overflow
    let sum = a.wrapping_add(b);

    // ถ้า overflow เกิดขึ้น ผลรวมจะน้อยกว่า *ทั้งสอง* ตัวตั้ง โดยไม่มี
    // overflow แน่นอนว่ามันจะต้องมีค่าอย่างน้อยเท่ากับตัวใดตัวหนึ่ง ดังนั้น เรามาเลือกสักตัว
    // แล้วตรวจสอบ
    if sum >= a {
        // การบวกไม่ล้น
        // **แบบฝึกหัด 08.1**: เขียนโค้ดเพื่อจัดการการบวกตัวทดในกรณีนี้
        unimplemented!()
    } else {
        // มิฉะนั้น การบวก *ล้นจริง* เป็นไปไม่ได้ที่การบวกตัวทดเข้าไป
        // จะล้นอีกครั้ง เพราะเราแค่บวก 0 หรือ 1
        (sum + if carry { 1 } else { 0 }, true)
    }
}
```

`overflow_add` เป็นฟังก์ชันที่ซับซ้อนพอสมควรจนสมเหตุสมผลที่จะมีกรณีทดสอบ สิ่งนี้ควรช่วยคุณตรวจสอบคำตอบของแบบฝึกหัดของคุณด้วย

```rust
/*#[test]*/
fn test_overflowing_add() {
    assert_eq!(overflowing_add(10, 100, false), (110, false));
    assert_eq!(overflowing_add(10, 100, true), (111, false));
    assert_eq!(overflowing_add(1 << 63, 1 << 63, false), (0, true));
    assert_eq!(overflowing_add(1 << 63, 1 << 63, true), (1, true));
    assert_eq!(overflowing_add(1 << 63, (1 << 63) - 1, true), (0, true));
}
```

## ชนิดข้อมูลสมทบ (Associated Types)

ตอนนี้เราพร้อมแล้วที่จะเขียนฟังก์ชันการบวกสำหรับ `BigInt` อย่างที่คุณอาจจะเดาได้ ตัวดำเนินการ `+` ถูกผูกไว้กับเทรต (`std::ops::Add`) ซึ่งเรากำลังจะอิมพลีเมนท์ให้กับ `BigInt`

โดยทั่วไป การบวกไม่จำเป็นต้องเป็นชนิดเดียวกัน (homogeneous): คุณสามารถบวกสิ่งของต่างชนิดกันได้ เช่น เวกเตอร์และจุด ดังนั้นเมื่ออิมพลีเมนท์ `Add` สำหรับชนิดข้อมูลหนึ่ง เราต้องระบชนิดข้อมูลของตัวถูกดำเนินการอีกตัวด้วย ในกรณีนี้ มันจะเป็น `BigInt` เหมือนกัน (และเราสามารถละไว้ได้ เพราะนั่นเป็นค่าตั้งต้น)

```rust
impl ops::Add<BigInt> for BigInt {
```

นอกจากฟังก์ชัน static และเมธอดแล้ว เทรตยังสามารถประกอบไปด้วย **ชนิดข้อมูลสมทบ (associated types)**: นี่คือชนิดข้อมูลที่ถูกเลือกโดยทุกการอิมพลีเมนท์เฉพาะเจาะจงของเทรต เมธอดของเทรตสามารถอ้างถึงชนิดข้อมูลนั้นได้ ในกรณีของการบวก มันถูกใช้เพื่อระบุชนิดข้อมูลของผลลัพธ์ (ดูเพิ่มเติมในเอกสารของ `Add`)

โดยทั่วไป คุณสามารถพิจารณา `BigInt` สองตัวที่ระบุข้างต้น (ในบรรทัด `impl`) เป็นชนิดข้อมูลขาเข้า (input types) ของการค้นหาเทรต: เมื่อ `a + b` ถูกเรียกใข้โดยที่ `a` มีชนิดข้อมูล `T` และ `b` มีชนิดข้อมูล `U` ทาง Rust จะพยายามหาการอิมพลีเมนท์ของ `Add` สำหรับ `T` ที่ซึ่งชนิดข้อมูลทางขวาคือ `U` ในทางกลับกัน Associated types คือชนิดข้อมูลขาออก (output types): สำหรับทุกการรวมตัวของชนิดข้อมูลขาเข้า จะมีชนิดข้อมูลผลลัพธ์หนึ่งตัวที่ถูกเลือกโดยการอิมพลีเมนท์ของ `Add` ที่สอดคล้องกัน

ที่นี่ เราเลือกชนิดข้อมูลผลลัพธ์ให้เป็น `BigInt` อีกครั้ง

```rust
    type Output = BigInt;
```

ตอนนี้เราสามารถเขียนฟังก์ชันจริงๆ ที่ทำการบวกได้

```rust
    fn add(self, rhs: BigInt) -> Self::Output {
        // เรารู้ว่าผลลัพธ์จะยาว *อย่างน้อย* เท่ากับตัวที่ยาวกว่าของตัวถูกดำเนินการทั้งสอง
        // ดังนั้นเราสามารถสร้างเวกเตอร์ที่มีความจุเพียงพอเพื่อหลีกเลี่ยงการจองพื้นที่ใหม่ที่ราคาแพง
        let max_len = cmp::max(self.data.len(), rhs.data.len());
        let mut result_vec: Vec<u64> = Vec::with_capacity(max_len);
        let mut carry = false; /* บิตตัวทดปัจจุบัน */
        for i in 0..max_len {
            let lhs_val = if i < self.data.len() { self.data[i] } else { 0 };
            let rhs_val = if i < rhs.data.len() { rhs.data[i] } else { 0 };
            // คำนวณเลขโดดถัดไปและตัวทด จากนั้น เก็บเลขโดดสำหรับผลลัพธ์ และตัวทด
            // ไว้ใช้ทีหลัง สังเกตว่าเราสามารถตั้งชื่อให้กับส่วนประกอบสองส่วนของคู่ลำดับที่
            // overflowing_add คืนมาได้อย่างไร
            let (sum, new_carry) = overflowing_add(lhs_val, rhs_val, carry);
            result_vec.push(sum);
            carry = new_carry;
        }
        // **แบบฝึกหัด 08.2**: จัดการ `carry` สุดท้าย และคืนค่าผลรวม
        unimplemented!()
    }
}
```

## เทรตและชนิดข้อมูลอ้างอิง (Traits and reference types)

ถ้าคุณตรวจสอบฟังก์ชันการบวกข้างต้นอย่างใกล้ชิด คุณจะสังเกตเห็นว่าจริงๆ แล้วมันบริโภคความเป็นเจ้าของของทั้งสองตัวถูกดำเนินการเพื่อสร้างผลลัพธ์ แน่นอนว่า โดยทั่วไปนี่ไม่ใช่สิ่งที่เราต้องการ เราอยากจะสามารถบวก `&BigInt` สองตัวเข้าด้วยกันมากกว่า

การเขียนสิ่งนี้ออกมาจะน่าเบื่อหน่ายนิดหน่อย เพราะการอิมพลีเมนท์เทรต (ไม่เหมือนฟังก์ชัน) ต้องการการระบุ lifetimes อย่างชัดเจนทั้งหมด ให้แน่ใจว่าคุณเข้าใจว่านิยามต่อไปนี้พูดว่าอะไร สังเกตว่าเราสามารถอิมพลีเมนท์เทรตให้กับชนิดข้อมูลอ้างอิง (reference type) ได้!

```rust
impl<'a, 'b> ops::Add<&'a BigInt> for &'b BigInt {
    type Output = BigInt;
    fn add(self, rhs: &'a BigInt) -> Self::Output {
        // **แบบฝึกหัด 08.3**: อิมพลีเมนท์ฟังก์ชันนี้
        unimplemented!()
    }
}
```

> **แบบฝึกหัด 08.4**: อิมพลีเมนท์การรวมตัวของอาร์กิวเมนต์ที่หายไปอีกสองแบบสำหรับ `Add` คุณไม่ควรต้องทำซ้ำการอิมพลีเมนท์ (duplicate code)

## โมดูล (Modules)

อย่างที่คุณได้เรียนรู้ การทดสอบสามารถเขียนได้ท่ามกลางการพัฒนาของคุณใน Rust เลย อย่างไรก็ตาม ถือว่าเป็นสไตล์ที่ดีที่จะรวบรวมการทดสอบทั้งหมดไว้ด้วยกัน สิ่งนี้มีประโยชน์โดยเฉพาะในกรณีที่คุณเขียนฟังก์ชันอรรถประโยชน์สำหรับการทดสอบ ที่โค้ดอื่นไม่ควรเรียกใช้

Rust เรียกกลุ่มของการนิยามที่ถูกจัดไว้ด้วยกันว่า **โมดูล (module)** คุณสามารถใส่การทดสอบในโมดูลย่อยได้ดังนี้ Attribute `cfg` ควบคุมว่าโมดูลนี้จะถูกคอมไพล์หรือไม่: ถ้าเราเพิ่มฟังก์ชันบางอย่างที่เป็นประโยชน์ต่อการทดสอบ Rust จะไม่เสียเวลาคอมไพล์พวกมันเมื่อคุณแค่ build โปรแกรมของคุณสำหรับการใช้งานปกติ นอกเหนือจากนั้น การทดสอบทำงานตามปกติ

```rust
#[cfg(test)]
mod tests {
    use part05::BigInt;

    /*#[test]*/
    fn test_add() {
        let b1 = BigInt::new(1 << 32);
        let b2 = BigInt::from_vec(vec![0, 1]);

        assert_eq!(&b1 + &b2, BigInt::from_vec(vec![1 << 32, 1]));
        // **แบบฝึกหัด 08.5**: เพิ่มกรณีอื่นๆ ให้กับการทดสอบนี้
    }
}
```

อย่างที่กล่าวไปแล้ว ภายนอกโมดูล เฉพาะ items ที่ถูกประกาศเป็นสาธารณะด้วย `pub` เท่านั้นที่อาจถูกใช้ได้ โมดูลย่อยสามารถเข้าถึงทุกสิ่งที่นิยามในโมดูลพ่อแม่ของพวกมัน ตัวโมดูลเองก็ถูกซ่อนจากภายนอกเป็นค่าตั้งต้น และสามารถทำให้เป็นสาธารณะได้ด้วย `pub` เมื่อคุณใช้ identifier (หรือ ทั่วไปกว่านั้น path เช่น `mod1::submod::name`) มันจะถูกตีความว่าสัมพัทธ์ (relative) กับโมดูลปัจจุบัน ดังนั้น ตัวอย่างเช่น เพื่อเข้าถึง `overflowing_add` จากภายใน `my_mod` คุณจะต้องระบุ path ที่ชัดเจนขึ้นโดยการเขียน `super::overflowing_add` ซึ่งบอก Rust ให้มองหาในโมดูลพ่อแม่

คุณสามารถทำให้ชื่อจากโมดูลอื่นพร้อมใช้งานแบบ local ได้ด้วย `use` โดยค่าตั้งต้น `use` ทำงานแบบ global ดังนั้นเช่น `use std;` นำเข้าชื่อ global `std` โดยการเติม `super::` หรือ `self::` ไปที่จุดเริ่มต้นของ path คุณทำให้มันสัมพัทธ์กับพ่อแม่หรือโมดูลปัจจุบันตามลำดับ (คุณยังสามารถสร้าง absolute path อย่างชัดแจ้งได้โดยเริ่มด้วย `::` เช่น `::std::cmp::min`) คุณสามารถพูดว่า `pub use path;` เพื่อนำเข้าชื่อและทำให้พวกมันพร้อมใช้งานแบบสาธารณะกับคนอื่นไปพร้อมๆ กัน สุดท้าย คุณสามารถนำเข้า public items ทั้งหมดของโมดูลทีเดียวได้ด้วย `use module::*;`

โมดูลสามารถถูกแยกไปไว้ในไฟล์แยกต่างหากได้ด้วยไวยากรณ์ `mod name;` เพื่ออธิบายสิ่งนี้ ขอผมออกนอกเส้นทางนิดหน่อยไปดูกระบวนการคอมไพล์ของ Rust ทาง Cargo เริ่มต้นโดยการเรียก `rustc` บนไฟล์ `src/lib.rs` หรือ `src/main.rs` ขึ้นอยู่กับว่าคุณคอมไพล์แอปพลิเคชันหรือไลบรารี เมื่อ `rustc` เจอ `mod name;` มันจะมองหาไฟล์ `name.rs` และ `name/mod.rs` และไปคอมไพล์ต่อที่นั่น (ถือเป็นข้อผิดพลาดถ้ามีทั้งสองไฟล์) คุณสามารถคิดว่าเนื้อหาของไฟล์ถูกฝังอยู่ที่จุดนี้ อย่างไรก็ตาม เฉพาะไฟล์ที่การคอมไพล์เริ่มต้นขึ้น และไฟล์ `name/mod.rs` เท่านั้นที่สามารถโหลดโมดูลจากไฟล์อื่นได้ สิ่งนี้ทำให้มั่นใจว่าโครงสร้างไดเรกทอรีสะท้อนโครงสร้างของโมดูล โดยที่ `mod.rs`, `lib.rs` และ `main.rs` เป็นตัวแทนของไดเรกทอรีหรือเครตนั้นๆ เอง (คล้ายกับ เช่น `__init__.py` ใน Python)
