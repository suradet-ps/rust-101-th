# Rust-101, ส่วนที่ 13: Concurrency (การทำงานพร้อมกัน), Arc, Send

```rust
use std::io::prelude::*;
use std::{io, fs, thread};
use std::sync::mpsc::{sync_channel, SyncSender, Receiver};
use std::sync::Arc;
```

จุดแวะถัดไปของเราคือฟีเจอร์การทำงานพร้อมกัน (concurrency features) ของ Rust เราจะเขียน "grep" เวอร์ชันเล็กของเราเอง เรียกว่า rgrep และมันจะทำงานสามอย่างพร้อมกัน: เธรดหนึ่งอ่านไฟล์นำเข้า, เธรดหนึ่งทำการจับคู่ (matching) จริงๆ, และอีกเธรดหนึ่งเขียนผลลัพธ์ ผมได้กล่าวไปแล้วในตอนต้นของคอร์สว่าระบบชนิดข้อมูลของ Rust (พูดให้ชัดคือ ระเบียบวินัยของความเป็นเจ้าของและการยืม) จะช่วยเราหลีกเลี่ยงหลุมพรางทั่วไปของการเขียนโปรแกรมแบบ concurrent: data races (การแย่งชิงข้อมูล) เราจะมาดูกันว่ามันทำงานอย่างไร

ก่อนที่เราจะไปถึงโค้ดจริงๆ เรานิยามโครงสร้างข้อมูล `Options` เพื่อเก็บรักษาข้อมูลทั้งหมดที่เราต้องใช้เพื่อทำงานให้สำเร็จ: จะทำงานกับไฟล์ไหน, จะมองหารูปแบบ (pattern) อะไร, และจะแสดงผลลัพธ์อย่างไร นอกจากการแค่พิมพ์บรรทัดที่ตรงกันทั้งหมดแล้ว เราจะเสนอให้มีการนับจำนวนพวกมัน หรืออีกทางเลือกคือเรียงลำดับพวกมันก่อนด้วย

```rust
#[derive(Clone, Copy)]
pub enum OutputMode {
    Print,
    SortAndPrint,
    Count,
}
use self::OutputMode::*;

pub struct Options {
    pub files: Vec<String>,
    pub pattern: String,
    pub output_mode: OutputMode,
}
```

ตอนนี้เราสามารถเขียนฟังก์ชันสามฟังก์ชันเพื่อทำงานจริงๆ ของการอ่าน, การจับคู่, และการพิมพ์ ตามลำดับ เพื่อที่จะเอาข้อมูลจากเธรดหนึ่งไปยังเธรดถัดไป เราจะใช้การส่งข้อความ (message passing): เราจะสร้างช่องทางการสื่อสาร (communication channels) ระหว่างเธรด โดยที่เธรดหนึ่งส่งข้อมูล และอีกเธรดหนึ่งรับมัน `SyncSender<T>` คือชนิดข้อมูลของปลายทางฝั่งส่งของช่องทางแบบประสานเวลา (synchronous channel) ที่ส่งข้อมูลชนิดข้อมูล `T` คำว่า Synchronous (ประสานเวลา) ที่นี่หมายความว่าการปฏิบัติการ `send` (ส่ง) สามารถบล็อก (block) ได้ โดยรอให้อีกฝั่งหนึ่งมีความคืบหน้า เราไม่อยากจะลงเอยด้วยการที่ทั้งไฟล์ถูกเก็บอยู่ในบัฟเฟอร์ของช่องทาง และผลลัพธ์ออกมาไม่ทันกับความเร็วของการอ่าน

เรายังจำเป็นต้องให้ทุกเธรดเข้าถึงตัวเลือก (options) ของงานที่พวกมันต้องทำ เนื่องจากคงค่อนข้างไม่จำเป็นที่จะคัดลอกตัวเลือกเหล่านี้ไปทั่ว เราจะใช้การนับ reference (reference-counting) เพื่อแบ่งปันมันระหว่างทุกเธรด `Arc` คือเวอร์ชันที่ปลอดภัยต่อเธรด (thread-safe) ของ `Rc` โดยใช้การปฏิบัติการแบบ atomic เพื่อเก็บรักษาตัวนับ reference

## ฟังก์ชันของทั้งสามเธรด (The Three Thread Functions)

ฟังก์ชันแรกอ่านไฟล์ และส่งทุกบรรทัดผ่าน `out_channel`

```rust
fn read_files(options: Arc<Options>, out_channel: SyncSender<String>) {
    for file in options.files.iter() {
        // ก่อนอื่น เราเปิดไฟล์ โดยละเลยข้อผิดพลาดใดๆ
        let file = fs::File::open(file).unwrap();
        // จากนั้นเราขอ `BufReader` สำหรับมัน ซึ่งจัดเตรียมฟังก์ชัน `lines` ให้
        let file = io::BufReader::new(file);
        for line in file.lines() {
            let line = line.unwrap();
            // ตอนนี้เราส่งบรรทัดผ่านช่องทาง โดยละเลยความเป็นไปได้ที่ `send` จะล้มเหลว
            out_channel.send(line).unwrap();
        }
    }
    // เมื่อเรา drop `out_channel` มันจะถูกปิด ซึ่งปลายทางอีกฝั่งสามารถสังเกตเห็นได้
}
```

ฟังก์ชันที่สองกรองบรรทัดที่มันได้รับผ่าน `in_channel` ด้วยรูปแบบ (pattern) และส่งสิ่งที่ตรงกันผ่าน `out_channel`

```rust
fn filter_lines(options: Arc<Options>,
                in_channel: Receiver<String>,
                out_channel: SyncSender<String>) {
    // เราสามารถวนซ้ำบนช่องทางได้ง่ายๆ ซึ่งจะหยุดเมื่อช่องทางถูกปิด
    for line in in_channel.iter() {
        // `contains` ทำงานกับชนิดข้อมูลของรูปแบบได้หลายแบบ แต่โดยเฉพาะอย่างยิ่ง เราสามารถใช้มันตรวจสอบ
        // ว่าสตริงหนึ่งถูกบรรจุอยู่ในอีกอันหนึ่งหรือไม่ นี่เป็นอีกตัวอย่างของ Rust ที่ใช้เทรต
        // เป็นตัวแทนของ overloading
        if line.contains(&options.pattern) {
            out_channel.send(line).unwrap();
        }
    }
}
```

ฟังก์ชันที่สามดำเนินการปฏิบัติการผลลัพธ์ โดยรับบรรทัดที่เกี่ยวข้องทาง `in_channel` ของมัน

```rust
fn output_lines(options: Arc<Options>, in_channel: Receiver<String>) {
    match options.output_mode {
        Print => {
            // ที่นี่ เราแค่พิมพ์ทุกบรรทัดที่เราเห็น
            for line in in_channel.iter() {
                println!("{}", line);
            }
        },
        Count => {
            // เราถูกคาดหวังให้นับจำนวนของบรรทัดที่ตรงกัน มี iterator adapter
            // ที่สะดวกสบายที่เราสามารถใช้กับงานนี้ได้
            let count = in_channel.iter().count();
            println!("{} hits for {}.", count, options.pattern);
        },
        SortAndPrint => {
            // เราถูกขอให้เรียงลำดับบรรทัดที่ตรงกันก่อนพิมพ์ ดังนั้นมารวบรวมพวกมันทั้งหมด
            // ในเวกเตอร์ local...
            let mut data: Vec<String> = in_channel.iter().collect();
            // ...และอิมพลีเมนท์การเรียงลำดับจริงๆ ทีหลัง
            unimplemented!()
        }
    }
}
```

## ฟังก์ชัน Run หลัก (The Main Run Function)

ด้วยการปฏิบัติการของทั้งสามเธรดที่ถูกนิยามแล้ว ตอนนี้เราสามารถอิมพลีเมนท์ฟังก์ชันที่ทำการ grep ตามตัวเลือกที่ให้มาได้

```rust
pub fn run(options: Options) {
    // เราย้าย `options` เข้าไปใน `Arc` เนื่องจากนั่นคือสิ่งที่คนงานเธรด (thread workers) คาดหวัง
    let options = Arc::new(options);

    // สิ่งนี้ตั้งค่าช่องทาง เราใช้ `sync_channel` ที่มีขนาดบัฟเฟอร์ 16 เพื่อหลีกเลี่ยง
    // การเติม RAM โดยไม่จำเป็น
    let (line_sender, line_receiver) = sync_channel(16);
    let (filtered_sender, filtered_receiver) = sync_channel(16);

    // Spawn เธรดอ่าน: `thread::spawn` รับ closure ที่ถูกรันในเธรดใหม่
    // คีย์เวิร์ด `move` บอก Rust อีกครั้งว่าเราต้องการให้ความเป็นเจ้าของของตัวแปรที่ถูกจับ
    // ถูกย้ายเข้าไปใน closure นี่หมายความว่าเราจำเป็นต้องทำการโคลนก่อน ไม่อย่างนั้นเราจะเสีย
    // options ของเราให้กับเธรดใหม่!
    let options1 = options.clone();
    let handle1 = thread::spawn(move || read_files(options1, line_sender));

    // เหมือนกันกับเธรดตัวกรอง
    let options2 = options.clone();
    let handle2 = thread::spawn(move || {
        filter_lines(options2, line_receiver, filtered_sender)
    });

    // และเธรดผลลัพธ์
    let options3 = options.clone();
    let handle3 = thread::spawn(move || output_lines(options3, filtered_receiver));

    // สุดท้าย รอจนกว่าทั้งสามเธรดทำงานของพวกมันเสร็จ การ join เธรดคือการรอให้
    // การทำงานสิ้นสุด (termination) สิ่งนี้สามารถล้มเหลวได้ถ้าเธรดนั้น panic: ในกรณีนี้ เราจะได้เข้าถึง
    // ข้อมูลที่มันมอบให้กับ `panic!` ที่นี่ เราแค่ยืนยัน (assert) ว่าพวกมันไม่ได้ panic -
    // เพื่อที่เราจะได้ panic ตัวเราเองถ้าคำยืนยันนั้นล้มเหลว
    handle1.join().unwrap();
    handle2.join().unwrap();
    handle3.join().unwrap();
}
```

ตอนนี้เรามีชิ้นส่วนทั้งหมดครบแล้วเพื่อทดสอบ rgrep ของเรากับตัวเลือกแบบ hard-coded บางอย่าง เราจำเป็นต้องเรียก `to_string` บน string literals เพื่อแปลงพวกมันเป็น `String` แบบ owned เต็มตัว

```rust
pub fn main() {
    let options = Options {
        files: vec!["src/part10.rs".to_string(),
                    "src/part11.rs".to_string(),
                    "src/part12.rs".to_string()],
        pattern: "let".to_string(),
        output_mode: Print
    };
    run(options);
}
```

> **แบบฝึกหัด 13.1**: เปลี่ยน rgrep ให้มันพิมพ์ไม่เพียงแค่บรรทัดที่ตรงกัน แต่ยังรวมถึงชื่อของไฟล์และหมายเลขของบรรทัดในไฟล์ด้วย คุณจะต้องเปลี่ยนชนิดข้อมูลของช่องทางจาก `String` เป็นบางอย่างที่บันทึกข้อมูลพิเศษนี้

## ความเป็นเจ้าของ, การยืม, และ Concurrency (Ownership, Borrowing, and Concurrency)

การสาธิตเล็กๆ ข้างต้นแสดงให้เห็นว่า concurrency ใน Rust มี API ที่ค่อนข้างเรียบง่าย เมื่อพิจารณาว่า Rust มี closures นั่นก็ไม่น่าแปลกใจทั้งหมด อย่างไรก็ตาม อย่างที่ปรากฏออกมา Rust ไปไกลกว่านั้นและรับประกันจริงๆ ถึงการไม่มี data races

Data race โดยทั่วไปถูกนิยามว่าเป็นการเข้าถึงตำแหน่งหน่วยความจำเดียวกันพร้อมกันแบบไม่ประสานเวลา (concurrent, unsynchronized accesses) โดยที่มีอย่างน้อยหนึ่งการเข้าถึงเป็นการเขียน (write) พูดอีกอย่างคือ data race คือการแก้ไข (mutation) ในขณะที่มีการตั้งชื่อซ้ำซ้อน (aliasing) ซึ่ง Rust กำจัดออกไปได้อย่างน่าเชื่อถือ! ปรากฏว่ากลไกเดียวกันที่ทำให้โปรแกรมเธรดเดียวของเราปลอดภัยทางหน่วยความจำ และที่ป้องกันเราจากการทำให้ iterators ใช้การไม่ได้ ก็ยังช่วยรักษาความปลอดภัยโค้ดหลายเธรดของเราต่อ data races ด้วย ตัวอย่างเช่น สังเกตว่า `read_files` ส่ง `String` ไปยัง `filter_lines` ได้อย่างไร ณ รันไทม์ มีเพียง pointer ไปยังข้อมูลตัวอักษรเท่านั้นที่ถูกย้ายไปมาจริงๆ (เหมือนกับเมื่อ `String` ถูกส่งไปให้ฟังก์ชันด้วย full ownership) อย่างไรก็ตาม `read_files` ต้องสละความเป็นเจ้าของของสตริงเพื่อดำเนินการ `send` ดังนั้นจึงเป็นไปไม่ได้ที่สตริงจะยังคงถูกยืมอยู่ หลังจากที่มันส่งสตริงไปให้อีกฝั่งแล้ว `read_files` ไม่มี pointer เข้าไปในเนื้อหาสตริงอีกต่อไป และดังนั้นจึงไม่มีทางที่จะ race บนข้อมูลกับคนอื่น

มีอะไรมากกว่านี้นิดหน่อย จำขอบเขต `'static` ที่เราต้องเติมให้กับ `register` ในส่วนก่อนหน้านี้ได้ไหม เพื่อให้แน่ใจว่า callbacks ไม่ได้อ้างถึง pointers ใดๆ ที่อาจจะกลายเป็นใช้การไม่ได้? นี่สำคัญพอๆ กันสำหรับการ spawn เธรด: โดยทั่วไป เธรดนั้นสามารถอยู่ได้นานกว่า stack frame ปัจจุบันมาก ดังนั้น มันต้องไม่ใช้ pointers ใดๆ ไปยังข้อมูลใน stack frame นั้น สิ่งนี้บรรลุผลได้โดยต้องการให้ `FnOnce` closure ที่ส่งไปให้ `thread::spawn` ใช้งานได้สำหรับ lifetime `'static` ดังที่คุณเห็นได้ในเอกสารของมัน สิ่งนี้หลีกเลี่ยง data race อีกชนิดหนึ่ง ที่ซึ่งการเข้าถึงของเธรด race กับการที่ผู้เรียก (callee) คืนหน่วยความจำ stack frame ของมัน ต้องขอบคุณแนวคิดเรื่อง lifetimes เท่านั้นที่ทำให้สิ่งนี้สามารถแสดงออกมาเป็นส่วนหนึ่งของชนิดข้อมูลของ `spawn` ได้

## Send

อย่างไรก็ตาม เรื่องราวยังไปไกลกว่านั้น ผมบอกข้างต้นว่า `Arc` เป็นเวอร์ชันปลอดภัยต่อเธรดของ `Rc` ซึ่งใช้การปฏิบัติการ atomic เพื่อจัดการตัวนับ reference จึงเป็นเรื่องสำคัญมากที่เราไม่ใช้ `Rc` ข้ามหลายเธรด ไม่อย่างนั้นตัวนับ reference อาจจะใช้การไม่ได้ และจริงๆ แล้ว ถ้าคุณแทนที่ `Arc` ด้วย `Rc` (และเพิ่ม imports ที่เหมาะสม) Rust จะบอกคุณว่ามีบางอย่างผิดปกติ เยี่ยมเลย แน่นอน แต่ว่ามันทำอย่างนั้นได้อย่างไร?

คำตอบถูกบอกใบ้อยู่แล้วใน error: มันจะพูดอะไรบางอย่างเกี่ยวกับ `Send` คุณอาจจะสังเกตเห็นว่า closure ใน `thread::spawn` ไม่ได้มีแค่ขอบเขต `'static` แต่ยังต้องผ่านเงื่อนไข `Send` ด้วย `Send` เป็นเทรต และเหมือนกับ `Copy` มันเป็นแค่ตัวแปะป้าย (marker) - ไม่มีฟังก์ชันที่จัดเตรียมโดย `Send` สิ่งที่เทรตบอกคือชนิดข้อมูลที่เป็น `Send` สามารถถูกส่งไปยังเธรดอื่นได้อย่างปลอดภัยโดยไม่ก่อปัญหา แน่นอนว่า data-types พื้นฐานทั้งหมดเป็น `Send` เช่นเดียวกับ `Arc` ซึ่งเป็นเหตุผลที่ Rust ยอมรับโค้ดของเรา แต่ `Rc` ไม่ใช่ `Send` และด้วยเหตุผลที่ดี! ถ้าเรามี `Rc` สองตัวไปยังข้อมูลเดียวกัน และส่งหนึ่งในนั้นไปยังเธรดอื่น ความยุ่งเหยิงอาจจะเกิดขึ้นเนื่องจากการขาดการประสานเวลา (synchronization)

ทีนี้ `Send` ในฐานะเทรตนั้นค่อนข้างพิเศษ มันมีสิ่งที่เรียกว่า **การอิมพลีเมนท์ตั้งต้น (default implementation)** นี่หมายความว่าทุกชนิดข้อมูลอิมพลีเมนท์ `Send` ยกเว้นว่ามันเลือกที่จะไม่รับ (opts out) การเลือกไม่รับนั้นแพร่เชื้อ (viral): ถ้าชนิดข้อมูลของคุณประกอบด้วยชนิดข้อมูลที่เลือกไม่รับ คุณก็จะไม่มี `Send` ด้วยเช่นกัน ดังนั้นถ้าสภาพแวดล้อมของ closure ของคุณประกอบด้วย `Rc` มันจะไม่เป็น `Send` ป้องกันไม่ให้มันก่อปัญหา อย่างไรก็ตาม ถ้าตัวแปรที่ถูกจับทุกตัวเป็น `Send` ทั้งสภาพแวดล้อมก็จะเป็นด้วย และคุณก็ผ่านฉลุย
