# Rust-101, ส่วนที่ 10: Closures

```rust
use std::fmt;
use part05::BigInt;
```

สมมติว่าเราต้องการเขียนฟังก์ชันที่ทำอะไรบางอย่างกับ, สมมติว่า, ทุกเลขโดดของ `BigInt` เราก็จะต้องมีวิธีการแสดงการกระทำที่เราต้องการให้ทำ และส่งสิ่งนี้ไปให้ฟังก์ชันของเรา ใน Rust ความพยายามแรกที่เป็นธรรมชาติในการแสดงสิ่งนี้คือการมีเทรตสำหรับมัน

ดังนั้น มานิยามเทรตที่เรียกร้องว่าชนิดข้อมูลต้องจัดเตรียมเมธอด `do_action` บนเลขโดด สิ่งนี้ทำให้เกิดคำถามทันที: เราจะส่ง `self` ไปยังฟังก์ชันนั้นอย่างไร? แบบเป็นเจ้าของ (Owned), แบบ reference แบ่งปัน (shared reference), หรือแบบ reference แก้ไขได้ (mutable reference)? กลยุทธ์ทั่วไปในการตอบคำถามนี้คือใช้ชนิดข้อมูลที่แข็งแรงที่สุดที่ยังคงทำงานได้ แน่นอนว่า การส่ง `self` ในรูปแบบ being owned ไม่เวิร์ก: เพราะฟังก์ชันจะบริโภค `self` และเราจะไม่สามารถเรียกมันได้อีกครั้ง บนเลขโดดตัวที่สอง ดังนั้นมาเลือกใช้ mutable reference กันเถอะ

```rust
trait Action {
    fn do_action(&mut self, digit: u64);
}
```

ตอนนี้เราสามารถเขียนฟังก์ชันที่รับ `a` บางตัวที่มีชนิดข้อมูล `A` โดยที่ชนิดข้อมูลนั้นเราสามารถเรียก `do_action` บน `a` และส่งเลขโดดทุกตัวให้มัน

```rust
impl BigInt {
    fn act_v1<A: Action>(&self, mut a: A) {
        // จำไว้ว่า `mut` ข้างบนเป็นแค่คำระบุบอก Rust ว่าเราโอเค
        // กับการที่ `a` ถูกแก้ไขค่า การเรียก `do_action` บน `a` รับ mutable reference ดังนั้นการแก้ไขค่า
        // จึงสามารถเกิดขึ้นได้จริงๆ
        for digit in self {
            a.do_action(digit);
        }
    }
}
```

เป็นขั้นตอนถัดไป เราต้องคิดการกระทำบางอย่างขึ้นมา และเขียนการอิมพลีเมนท์ `Action` ที่เหมาะสมสำหรับมัน ดังนั้น สมมติว่าเราต้องการพิมพ์เลขโดดทุกตัว และเพื่อทำให้มันน่าเบื่อน้อยลง เราต้องการให้เลขโดดถูกนำหน้าด้วยสตริงอะไรก็ได้สักตัว `do_action` จำเป็นต้องรู้สตริงนี้ ดังนั้นเราเก็บมันไว้ใน `self`

```rust
struct PrintWithString {
    prefix: String,
}

impl Action for PrintWithString {
    // ที่นี่เราทำการพิมพ์ prefix และเลขโดดจริงๆ เราไม่ได้ใช้ประโยชน์จากความสามารถ
    // ในการเปลี่ยน `self` ที่นี่ แต่เราสามารถแทนที่ prefix ได้ถ้าเราต้องการ
    fn do_action(&mut self, digit: u64) {
        println!("{}{}", self.prefix, digit);
    }
}
```

สุดท้าย ฟังก์ชันนี้รับ `BigInt` และ prefix และพิมพ์เลขโดดพร้อมกับ prefix ที่ให้มา มันทำอย่างนั้นโดยสร้าง instance ของ `PrintWithString` ให้ prefix กับมัน และจากนั้นส่งสิ่งนั้นไปให้ `act_v1` เนื่องจาก `PrintWithString` อิมพลีเมนท์ `Action` ทาง Rust จึงรู้แล้วว่าต้องทำอะไร

```rust
fn print_with_prefix_v1(b: &BigInt, prefix: String) {
    let my_action = PrintWithString { prefix: prefix };
    b.act_v1(my_action);
}
```

นี่คือฟังก์ชัน main เล็กๆ สาธิตโค้ดข้างต้นขณะทำงาน อย่าลืมแก้ไข `main.rs` เพื่อรันมัน

```rust
pub fn main() {
    let bignum = BigInt::new(1 << 63) + BigInt::new(1 << 16) + BigInt::new(1 << 63);
    print_with_prefix_v1(&bignum, "Digit: ".to_string());
}
```

## Closures

เอาล่ะ อย่างที่ปรากฏออกมา รูปแบบของการอธิบายรูปแบบของการกระทำ ที่สามารถถือข้อมูลเพิ่มเติมเข้ามาได้นี้ เป็นสิ่งที่พบบ่อยมาก โดยทั่วไป สิ่งนี้เรียกว่า **closure** Closures รับอาร์กิวเมนต์บางอย่างและผลิตผลลัพธ์ และพวกมันมีสภาพแวดล้อม (environment) ที่พวกมันสามารถใช้ได้ ซึ่งสอดคล้องกับชนิดข้อมูล `PrintWithString` (หรือชนิดข้อมูลอื่นใดที่อิมพลีเมนท์ `Action`) อีกครั้ง เรามีทางเลือกในการส่งสภาพแวดล้อมนี้ในรูปแบบ owned หรือ borrowed ดังนั้นจึงมีเทรตสามอย่างสำหรับ closures ใน Rust: `Fn` - closures ได้รับ shared reference, `FnMut` - closures ได้รับ mutable reference, และ `FnOnce` - closures บริโภคสภาพแวดล้อมของพวกมัน (และดังนั้นจึงสามารถถูกเรียกได้เพียงครั้งเดียว) ไวยากรณ์สำหรับเทรต closure ที่รับอาร์กิวเมนต์ชนิดข้อมูล `T1, T2, ...` และคืนค่าบางอย่างที่มีชนิดข้อมูล `U` คือ `Fn(T1, T2, ...) -> U`

สิ่งนี้ระบุ `act` คล้ายกับข้างต้นมาก แต่ตอนนี้เราเรียกร้องให้ `A` เป็นชนิดข้อมูลของ closure ที่แก้ไขค่าสภาพแวดล้อมที่ถูกยืมมาของมัน รับเลขโดดตัวหนึ่ง และไม่คืนค่าอะไรเลย

```rust
impl BigInt {
    fn act<A: FnMut(u64)>(&self, mut a: A) {
        for digit in self {
            // เราสามารถเรียก closures ราวกับว่าพวกมันเป็นฟังก์ชัน - แต่จริงๆ แล้ว สิ่งที่เกิดขึ้นที่นี่
            // ถูกแปลเป็นสิ่งที่เหมือนกับที่เราเขียนข้างต้น ใน `act_v1`
            a(digit);
        }
    }
}
```

ตอนนี้เราเห็นแล้วว่าเขียนฟังก์ชันที่ทำงานกับ closures อย่างไร มาดูกันว่าจะเขียน closure อย่างไร

```rust
pub fn print_with_prefix(b: &BigInt, prefix: String) {
    // ไวยากรณ์สำหรับ closures คือ `|arg1, arg2, ...| code` สังเกตว่า closure สามารถอ้างถึง
    // ตัวแปรอย่าง `prefix` ที่มันไม่ได้รับมาเป็นอาร์กิวเมนต์ - ตัวแปรที่บังเอิญ
    // ปรากฏอยู่ภายนอก closure เราเรียกว่า closure ทำการจับ (captures) ตัวแปร ตอนนี้ Rust จะ
    // สร้างชนิดข้อมูลอัตโนมัติ (เหมือน `PrintWithString`) สำหรับสภาพแวดล้อมของ closure
    // พร้อมฟิลด์สำหรับทุกตัวแปรที่ถูกจับ อิมพลีเมนท์เทรต closure สำหรับชนิดข้อมูลนี้ โดยที่
    // การกระทำที่ทำถูกกำหนดโดยโค้ดของ closure และสุดท้ายมันจะ
    // สร้าง instance ของชนิดข้อมูลสภาพแวดล้อมที่นี่ ที่จุดนิยามของ closure และเติมมัน
    // ด้วยค่าที่ถูกจับ
    b.act(|digit| println!("{}{}", prefix, digit));
}
```

คุณสามารถเปลี่ยน `main` เพื่อเรียกฟังก์ชันนี้ และคุณควรสังเกตเห็นว่า - ไม่มีอะไร, ไม่มีพฤติกรรมแตกต่างกัน แต่เราเขียนโค้ดที่เป็น boilerplate น้อยกว่ามาก!

จำได้ไหมที่เราตัดสินใจใช้เทรต `FnMut` ข้างต้น? นี่หมายความว่า closure ของเราสามารถแก้ไขค่าสภาพแวดล้อมของมันได้จริงๆ ตัวอย่างเช่น เราสามารถใช้สิ่งนั้นเพื่อนับเลขโดดในขณะที่พวกมันถูกพิมพ์

```rust
pub fn print_and_count(b: &BigInt) {
    let mut count: usize = 0;
    // ครั้งนี้ สภาพแวดล้อมจะประกอบด้วยฟิลด์ของชนิดข้อมูล `&mut usize` ที่จะถูก
    // กำหนดค่าเริ่มต้นด้วย mutable reference ของ `count` ตัว closure เอง เนื่องจากมันยืม
    // สภาพแวดล้อมของมันแบบแก้ไขได้ จึงสามารถเข้าถึงฟิลด์นี้และแก้ไข `count` ผ่านมันได้ เมื่อ `act` คืนค่า
    // closure จะถูกทำลาย และ `count` ก็จะไม่ถูกยืมอีกต่อไป เนื่องจาก closures ถูกคอมไพล์ลงมา
    // เป็นชนิดข้อมูลปกติ borrow checking ทั้งหมดจึงยังคงทำงานตามปกติ และเราไม่สามารถ
    // เผลอทำ closure หลุดรอดไปที่ไหนสักแห่งโดยที่ยังบรรจุ reference ที่ตายแล้ว
    // ไว้ในสภาพแวดล้อมของมัน
    b.act(|digit| { println!("{}: {}", count, digit); count = count + 1; });
    println!("There are {} digits", count);
}
```

## ความสนุกกับ Iterators และ Closures (Fun with iterators and closures)

ถ้าคุณคุ้นเคยกับภาษาเชิงฟังก์ชัน (functional languages) คุณอาจจะตระหนักได้ว่าเราสามารถมีความสนุกมากมายกับ iterators และ closures Rust จัดเตรียมเมธอดจำนวนมากบน iterators ที่อนุญาตให้เราเขียนการประมวลผล list ในสไตล์ functional ที่สวยงาม

สมมติว่าเราต้องการเขียนฟังก์ชันที่เพิ่มค่าทุกรายการของ `Vec` ด้วยตัวเลขบางตัว จากนั้นมองหาตัวเลขที่มากกว่าค่าขีดเริ่มเปลี่ยน (threshold) บางค่า และพิมพ์พวกมันออกมา

```rust
fn inc_print_threshold(v: &Vec<i32>, offset: i32, threshold: i32) {
    // `map` รับ closure ที่ถูกประยุกต์ใช้กับทุกสมาชิกของตัววนซ้ำ `filter` เอา
    // สมาชิกออกจากตัววนซ้ำที่ไม่ผ่านการทดสอบที่ให้โดย closure
    // เนื่องจาก closures ทั้งหมดนี้ถูกคอมไพล์ลงมาเป็นรูปแบบที่อธิบายข้างต้น จึงไม่มี
    // การจองพื้นที่ heap เกิดขึ้นที่นี่จริงๆ สิ่งนี้ทำให้ closures มีประสิทธิภาพมาก และมันทำให้การปรับแต่ง (optimization)
    // เป็นเรื่องเล็กน้อย: โค้ดผลลัพธ์จะดูเหมือนกับคุณเขียนลูปด้วยมือในภาษา C
    for i in v.iter().map(|n| *n + offset).filter(|n| *n > threshold) {
        println!("{}", i);
    }
}
```

บางครั้งมันก็มีประโยชน์ที่จะรู้ทั้งตำแหน่งของสมาชิกบางตัวในรายการ และค่าของมัน นั่นคือที่ที่ฟังก์ชัน `enumerate` ช่วยได้

```rust
fn print_enumerated<T: fmt::Display>(v: &Vec<T>) {
    // `enumerate` เปลี่ยนตัววนซ้ำบน `T` ให้เป็นตัววนซ้ำบน `(usize, T)` โดยที่สมาชิก
    // ตัวแรกแค่นับตำแหน่งในตัววนซ้ำ เราสามารถทำ pattern matching ได้เลยใน
    // หัวของลูปเพื่อระบุชื่อให้กับทั้งตำแหน่ง และค่า
    for (i, t) in v.iter().enumerate() {
        println!("Position {}: {}", i, t);
    }
}
```

และเป็นตัวอย่างสุดท้าย เรายังสามารถรวบรวม (collect) สมาชิกทั้งหมดของตัววนซ้ำ และใส่พวกมัน เช่น ลงในเวกเตอร์

```rust
fn filter_vec_by_divisor(v: &Vec<i32>, divisor: i32) -> Vec<i32> {
    // ที่นี่ ชนิดข้อมูลผลลัพธ์ของ `collect` ถูกอนุมาน (inferred) ตามชนิดข้อมูลผลลัพธ์ของฟังก์ชันของเรา
    // โดยทั่วไป มันสามารถคืนค่าอะไรก็ได้ที่อิมพลีเมนท์ `FromIterator` สังเกตว่า `iter` ให้
    // ตัววนซ้ำบน `i32` ที่ถูกยืมมาแก่เรา แต่เราต้องการเป็นเจ้าของพวกมันสำหรับผลลัพธ์ ดังนั้นเราเลยแทรก
    // `map` เพื่อ dereference
    v.iter().map(|n| *n).filter(|n| *n % divisor == 0).collect()
}
```
